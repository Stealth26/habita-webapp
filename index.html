import { useState, useEffect } from 'react';


export default function HabitaEstimator() {

  // Stati per i campi di input

  const [zona, setZona] = useState('');

  const [tipoImmobile, setTipoImmobile] = useState('');

  const [maxOspiti, setMaxOspiti] = useState(2);

  const [giorniDisponibili, setGiorniDisponibili] = useState(180);

  const [statoImmobile, setStatoImmobile] = useState('');

  const [email, setEmail] = useState('');

  const [telefono, setTelefono] = useState('');

  const [step, setStep] = useState(1);

  const [risultato, setRisultato] = useState(null);

  const [errore, setErrore] = useState('');

  const [province, setProvince] = useState([]);

  const [filteredProvince, setFilteredProvince] = useState([]);

  const [searchTerm, setSearchTerm] = useState('');

  const [dropdownOpen, setDropdownOpen] = useState(false);

  const [nomeUtente, setNomeUtente] = useState('');


  // Opzioni per i dropdown

  const tipiImmobile = ['Monolocale', 'Bilocale', 'Trilocale', 'Quadrilocale', 'Villa'];

  const statiImmobile = ['Arredato bene', 'Normale', 'Da sistemare'];


  // Lista delle province italiane

  useEffect(() => {

    const provinceItaliane = [

      'Agrigento', 'Alessandria', 'Ancona', 'Aosta', 'Arezzo', 'Ascoli Piceno', 'Asti', 'Avellino', 'Bari', 'Barletta-Andria-Trani',

      'Belluno', 'Benevento', 'Bergamo', 'Biella', 'Bologna', 'Bolzano', 'Brescia', 'Brindisi', 'Cagliari', 'Caltanissetta',

      'Campobasso', 'Carbonia-Iglesias', 'Caserta', 'Catania', 'Catanzaro', 'Chieti', 'Como', 'Cosenza', 'Cremona', 'Crotone',

      'Cuneo', 'Enna', 'Fermo', 'Ferrara', 'Firenze', 'Foggia', 'Forlì-Cesena', 'Frosinone', 'Genova', 'Gorizia',

      'Grosseto', 'Imperia', 'Isernia', 'La Spezia', 'L\'Aquila', 'Latina', 'Lecce', 'Lecco', 'Livorno', 'Lodi',

      'Lucca', 'Macerata', 'Mantova', 'Massa-Carrara', 'Matera', 'Messina', 'Milano', 'Modena', 'Monza e Brianza', 'Napoli',

      'Novara', 'Nuoro', 'Ogliastra', 'Olbia-Tempio', 'Oristano', 'Padova', 'Palermo', 'Parma', 'Pavia', 'Perugia',

      'Pesaro e Urbino', 'Pescara', 'Piacenza', 'Pisa', 'Pistoia', 'Pordenone', 'Potenza', 'Prato', 'Ragusa', 'Ravenna',

      'Reggio Calabria', 'Reggio Emilia', 'Rieti', 'Rimini', 'Roma', 'Rovigo', 'Salerno', 'Medio Campidano', 'Sassari', 'Savona',

      'Siena', 'Siracusa', 'Sondrio', 'Taranto', 'Teramo', 'Terni', 'Torino', 'Trapani', 'Trento', 'Treviso',

      'Trieste', 'Udine', 'Varese', 'Venezia', 'Verbano-Cusio-Ossola', 'Vercelli', 'Verona', 'Vibo Valentia', 'Vicenza', 'Viterbo'

    ];

    setProvince(provinceItaliane.sort());

    setFilteredProvince(provinceItaliane.sort());

  }, []);


  // Filtra le province in base alla ricerca

  useEffect(() => {

    if (searchTerm) {

      const filtered = province.filter(provincia => 

        provincia.toLowerCase().includes(searchTerm.toLowerCase())

      );

      setFilteredProvince(filtered);

    } else {

      setFilteredProvince(province);

    }

  }, [searchTerm, province]);


  // Funzione per avanzare al secondo step

  const iniziaSimulazione = () => {

    setStep(2);

  };


  // Calcola il risultato stimato con valori reali

  const calcolaRisultato = () => {

    if (!zona || !tipoImmobile || !statoImmobile) {

      setErrore('Compila tutti i campi per continuare');

      return;

    }

    

    setErrore('');

    

    // Valori reali per le principali province

    const baseRateMap = {

      'Milano': 120,

      'Roma': 110,

      'Firenze': 100,

      'Venezia': 130,

      'Napoli': 85,

      'Bologna': 95,

      'Torino': 85,

      'Rimini': 90,

      'Palermo': 80,

      'Bari': 75,

      'Catania': 70,

      'Verona': 90,

      'Genova': 85,

      'Monza e Brianza': 95,

    };

    

    // Valori per province non specificate esplicitamente

    const baseRatePerArea = {

      'Nord': 80,

      'Centro': 75,

      'Sud': 65,

      'Isole': 70

    };

    

    // Mappa delle aree

    const areaMap = {

      'Nord': [

        'Aosta', 'Torino', 'Milano', 'Bergamo', 'Brescia', 'Como', 'Varese', 'Verbano-Cusio-Ossola', 'Novara', 'Vercelli',

        'Biella', 'Monza e Brianza', 'Lecco', 'Lodi', 'Pavia', 'Cremona', 'Mantova', 'Sondrio', 'Trento', 'Bolzano', 'Venezia',

        'Padova', 'Rovigo', 'Vicenza', 'Verona', 'Treviso', 'Belluno', 'Udine', 'Pordenone', 'Gorizia', 'Trieste', 'Genova',

        'Savona', 'La Spezia', 'Imperia'

      ],

      'Centro': [

        'Bologna', 'Ferrara', 'Forlì-Cesena', 'Modena', 'Parma', 'Piacenza', 'Ravenna', 'Reggio Emilia', 'Rimini', 'Firenze',

        'Arezzo', 'Grosseto', 'Livorno', 'Lucca', 'Massa-Carrara', 'Pisa', 'Pistoia', 'Prato', 'Siena', 'Roma', 'Frosinone',

        'Latina', 'Rieti', 'Viterbo', 'Perugia', 'Terni', 'Ancona', 'Ascoli Piceno', 'Fermo', 'Macerata', 'Pesaro e Urbino'

      ],

      'Sud': [

        'Napoli', 'Avellino', 'Benevento', 'Caserta', 'Salerno', 'L\'Aquila', 'Chieti', 'Pescara', 'Teramo', 'Campobasso',

        'Isernia', 'Bari', 'Barletta-Andria-Trani', 'Brindisi', 'Foggia', 'Lecce', 'Taranto', 'Potenza', 'Matera', 'Cosenza',

        'Catanzaro', 'Crotone', 'Reggio Calabria', 'Vibo Valentia'

      ],

      'Isole': [

        'Palermo', 'Agrigento', 'Caltanissetta', 'Catania', 'Enna', 'Messina', 'Ragusa', 'Siracusa', 'Trapani', 'Cagliari',

        'Carbonia-Iglesias', 'Medio Campidano', 'Nuoro', 'Ogliastra', 'Olbia-Tempio', 'Oristano', 'Sassari'

      ]

    };

    

    // Determina il tasso base in base alla provincia

    let baseRate = baseRateMap[zona];

    if (!baseRate) {

      // Se la provincia non è nella mappa, cerca in quale area si trova

      for (const [area, province] of Object.entries(areaMap)) {

        if (province.includes(zona)) {

          baseRate = baseRatePerArea[area];

          break;

        }

      }

      // Se ancora non trova, usa un valore predefinito

      if (!baseRate) baseRate = 70;

    }

    

    const tipoMultiplier = {

      'Monolocale': 0.8,

      'Bilocale': 1,

      'Trilocale': 1.3,

      'Quadrilocale': 1.6,

      'Villa': 2.2

    }[tipoImmobile] || 1;

    

    const statoMultiplier = {

      'Arredato bene': 1.2,

      'Normale': 1,

      'Da sistemare': 0.8

    }[statoImmobile] || 1;

    

    const ospitiMultiplier = Math.min(maxOspiti * 0.15 + 0.85, 2.5);

    const giorniCalcolati = giorniDisponibili;

    const occupationRate = Math.min(giorniCalcolati / 365 * 0.9, 0.9);

    

    // Calcolo affitto breve termine

    const stimaGiornaliera = baseRate * tipoMultiplier * ospitiMultiplier * statoMultiplier;

    const stimaMensile = Math.round(stimaGiornaliera * 30 * occupationRate);

    const stimaAnnuale = Math.round(stimaGiornaliera * giorniCalcolati * occupationRate);

    

    // Calcolo affitto lungo termine

    // I valori sono basati su medie reali di affitto mensile

    const baseRateLungoTermineMap = {

      'Milano': 1200,

      'Roma': 1100,

      'Firenze': 950,

      'Venezia': 1000,

      'Napoli': 750,

      'Bologna': 900,

      'Torino': 750,

      'Rimini': 800,

      'Palermo': 650,

      'Bari': 600,

      'Catania': 600,

      'Verona': 800,

      'Genova': 700,

      'Monza e Brianza': 850,

    };

    

    const baseRateLungoTerminePerArea = {

      'Nord': 700,

      'Centro': 650,

      'Sud': 550,

      'Isole': 600

    };

    

    // Determina il tasso base per lungo termine

    let baseRateLungoTermine = baseRateLungoTermineMap[zona];

    if (!baseRateLungoTermine) {

      for (const [area, province] of Object.entries(areaMap)) {

        if (province.includes(zona)) {

          baseRateLungoTermine = baseRateLungoTerminePerArea[area];

          break;

        }

      }

      if (!baseRateLungoTermine) baseRateLungoTermine = 600;

    }

    

    const lungoTermineMultiplier = {

      'Monolocale': 0.7,

      'Bilocale': 1,

      'Trilocale': 1.3,

      'Quadrilocale': 1.6,

      'Villa': 2

    }[tipoImmobile] || 1;

    

    const stimaMensileLungoTermine = Math.round(baseRateLungoTermine * lungoTermineMultiplier * statoMultiplier);

    const stimaAnnualeLungoTermine = stimaMensileLungoTermine * 12;

    

    // Generate random name for testimonial

    const nomi = ['Marco', 'Laura', 'Simone', 'Giulia', 'Andrea', 'Chiara', 'Francesco', 'Valentina', 'Alessandro', 'Sofia'];

    const cognomi = ['B.', 'C.', 'D.', 'F.', 'G.', 'L.', 'M.', 'P.', 'R.', 'S.'];

    const randomNome = nomi[Math.floor(Math.random() * nomi.length)];

    const randomCognome = cognomi[Math.floor(Math.random() * cognomi.length)];

    

    setNomeUtente(`${randomNome} ${randomCognome} — ${zona}`);

    

    setRisultato({

      mensile: stimaMensile,

      annuale: stimaAnnuale,

      mensileLungoTermine: stimaMensileLungoTermine,

      annualeLungoTermine: stimaAnnualeLungoTermine

    });

    

    setStep(3);

  };


  // Funzione per prenotare una consulenza

  const prenotaConsulenza = () => {

    if (!email && !telefono) {

      setErrore('Inserisci almeno un contatto (email o telefono) per procedere');

      return;

    }

    

    setErrore('');

    setStep(4);

  };


  return (

    <div className="min-h-screen w-full bg-gray-50">

      {/* Logo header - presente in tutte le schermate */}

      <div className="p-4 flex items-center">

        <div className="text-2xl font-bold text-red-500">HABITA</div>

      </div>

      

      {step === 1 && (

        <div className="flex flex-col items-center justify-center min-h-screen w-full bg-red-500 text-white text-center px-4 pt-16">

          <h1 className="text-3xl font-bold mb-6">Quanto potrebbe renderti la tua casa con gli affitti brevi? 🏠💰</h1>

          <p className="text-xl mb-8">Scoprilo in 60 secondi con Habita Estimator</p>

          <button 

            onClick={iniziaSimulazione}

            className="w-full max-w-sm bg-white text-red-500 rounded-lg py-4 font-bold text-lg shadow-lg hover:bg-gray-100 transition duration-300"

          >

            Inizia la simulazione

          </button>

        </div>

      )}


      {step === 2 && (

        <div className="min-h-screen w-full pb-8">

          <div className="bg-red-500 text-white text-center py-4 px-4 mb-6">

            <h2 className="text-2xl font-bold">Stimatore di Rendita</h2>

            <p className="text-gray-200 mt-2">Completa il modulo per ottenere una stima personalizzata</p>

          </div>

          

          <div className="max-w-md mx-auto px-4 space-y-6">

            <div className="relative">

              <label className="block text-gray-700 mb-2 font-medium">🏙️ Provincia</label>

              <div className="relative">

                <input 

                  type="text" 

                  placeholder="Cerca la tua provincia..."

                  className="w-full p-4 border rounded-lg"

                  value={searchTerm}

                  onChange={(e) => {

                    setSearchTerm(e.target.value);

                    setDropdownOpen(true);

                  }}

                  onClick={() => setDropdownOpen(true)}

                />

                {dropdownOpen && (

                  <div className="absolute z-10 w-full mt-1 bg-white border rounded-lg shadow-lg max-h-60 overflow-y-auto">

                    {filteredProvince.map((provincia) => (

                      <div 

                        key={provincia} 

                        className="p-3 hover:bg-gray-100 cursor-pointer"

                        onClick={() => {

                          setZona(provincia);

                          setSearchTerm(provincia);

                          setDropdownOpen(false);

                        }}

                      >

                        {provincia}

                      </div>

                    ))}

                    {filteredProvince.length === 0 && (

                      <div className="p-3 text-gray-500">Nessuna provincia trovata</div>

                    )}

                  </div>

                )}

              </div>

            </div>

            

            <div>

              <label className="block text-gray-700 mb-2 font-medium">🏘️ Tipo di immobile</label>

              <select 

                className="w-full p-4 border rounded-lg"

                value={tipoImmobile}

                onChange={(e) => setTipoImmobile(e.target.value)}

              >

                <option value="">Seleziona il tipo di immobile</option>

                {tipiImmobile.map(tipo => (

                  <option key={tipo} value={tipo}>{tipo}</option>

                ))}

              </select>

            </div>

            

            <div>

              <label className="block text-gray-700 mb-2 font-medium">👥 Numero massimo di ospiti</label>

              <div className="flex items-center">

                <button 

                  className="bg-gray-200 px-4 py-3 rounded-l-lg font-bold text-xl"

                  onClick={() => setMaxOspiti(prev => Math.max(1, prev - 1))}

                >

                  -

                </button>

                <div className="flex-1 text-center py-3 border-t border-b font-medium text-lg">

                  {maxOspiti}

                </div>

                <button 

                  className="bg-gray-200 px-4 py-3 rounded-r-lg font-bold text-xl"

                  onClick={() => setMaxOspiti(prev => Math.min(20, prev + 1))}

                >

                  +

                </button>

              </div>

            </div>

            

            <div>

              <label className="block text-gray-700 mb-2 font-medium">📅 Giorni disponibili all'anno: {giorniDisponibili}</label>

              <input 

                type="range" 

                min="30" 

                max="365"

                value={giorniDisponibili}

                onChange={(e) => setGiorniDisponibili(parseInt(e.target.value))}

                className="w-full h-2 bg-gray-200 rounded-lg appearance-none cursor-pointer"

              />

              <div className="flex justify-between text-xs text-gray-500 mt-1">

                <span>30 giorni</span>

                <span>365 giorni</span>

              </div>

            </div>

            

            <div>

              <label className="block text-gray-700 mb-2 font-medium">🛋️ Stato dell'immobile</label>

              <select 

                className="w-full p-4 border rounded-lg"

                value={statoImmobile}

                onChange={(e) => setStatoImmobile(e.target.value)}

              >

                <option value="">Seleziona lo stato dell'immobile</option>

                {statiImmobile.map(stato => (

                  <option key={stato} value={stato}>{stato}</option>

                ))}

              </select>

            </div>

            

            {errore && <p className="text-red-500 text-sm text-center">{errore}</p>}

            

            <div className="flex justify-center">

              <button 

                onClick={calcolaRisultato}

                className="px-12 py-4 bg-red-500 text-white rounded-lg font-bold text-lg shadow-lg hover:bg-red-600 transition duration-300"

              >

                Calcola

              </button>

            </div>

          </div>

        </div>

      )}


      {step === 3 && risultato && (

        <div className="min-h-screen w-full pb-8">

          <div className="max-w-md mx-auto px-4">

            <div className="bg-white p-6 rounded-lg shadow-md mt-4">

              <h2 className="text-2xl font-bold mb-2">La tua stima:</h2>

              <p className="mb-3">Il tuo immobile potrebbe generare fino a</p>

              <div className="text-3xl font-bold mb-2">€{risultato.mensile} al mese</div>

              <p className="mb-6">con gli affitti brevi</p>

              

              <div className="flex items-end justify-between border-b pb-4 mb-4">

                <div className="text-center">

                  <div className="text-xl mb-1">€{risultato.mensileLungoTermine}/mese</div>

                  <div className="w-20 bg-gray-200" style={{height: `${(risultato.mensileLungoTermine / risultato.mensile) * 120}px`}}></div>

                  <div className="mt-2 text-sm">Affitto lungo termine</div>

                </div>

                

                <div className="text-center">

                  <div className="text-xl mb-1">€{risultato.mensile}/mese</div>

                  <div className="w-20 bg-red-500" style={{height: "120px"}}></div>

                  <div className="mt-2 text-sm">Affitto breve termine</div>

                </div>

              </div>

              

              <div className="mb-6">

                <div className="font-medium">{nomeUtente}</div>

                <p className="italic text-gray-600">"Non pensavo che un {tipoImmobile.toLowerCase()} potesse rendermi così tanto. Il team Habita ha fatto tutto al posto mio."</p>

              </div>

              

              <button 

                onClick={() => setStep(3.5)}

                className="w-full bg-red-500 text-white rounded-lg py-4 font-bold text-lg shadow-lg hover:bg-red-600 transition duration-300"

              >

                Si, voglio guadagnare con la mia casa

              </button>

            </div>

            

            {step === 3.5 && (

              <div className="bg-white p-6 rounded-lg shadow-md mt-4">

                <p className="text-lg font-medium text-center mb-4">

                  Vuoi un'analisi gratuita del tuo caso specifico?

                </p>

                

                <div className="space-y-4">

                  <input 

                    type="email" 

                    placeholder="Email *"

                    className="w-full p-4 border rounded-lg"

                    value={email}

                    onChange={(e) => setEmail(e.target.value)}

                  />

                  

                  <input 

                    type="tel" 

                    placeholder="Numero di telefono *"

                    className="w-full p-4 border rounded-lg"

                    value={telefono}

                    onChange={(e) => setTelefono(e.target.value)}

                  />

                  

                  <p className="text-sm text-gray-500 text-center">* Almeno uno dei due campi è obbligatorio</p>

                  

                  {errore && <p className="text-red-500 text-sm text-center">{errore}</p>}

                  

                  <button 

                    onClick={prenotaConsulenza}

                    className="w-full bg-red-500 text-white rounded-lg py-4 font-bold text-lg shadow-lg hover:bg-red-600 transition duration-300"

                  >
